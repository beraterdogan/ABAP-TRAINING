REPORT ZBE_P_002.

DATA: GV_ALAN     TYPE INT4,
      GV_CEVRE    TYPE INT4,
      GV_CONSTANT TYPE P DECIMALS 9 VALUE '0.4330127'.


SELECTION-SCREEN BEGIN OF BLOCK RAD1 WITH FRAME TITLE TEXT-001.

PARAMETERS: P_KARE   RADIOBUTTON GROUP GR1 USER-COMMAND US1,
            P_DIKDOR RADIOBUTTON GROUP GR1,
            P_UCGEN  RADIOBUTTON GROUP GR1.

SELECTION-SCREEN END OF BLOCK RAD1 .


SELECTION-SCREEN BEGIN OF BLOCK RAD2 WITH FRAME TITLE TEXT-002.

PARAMETERS: P_ALAN  AS CHECKBOX,
            P_CEVRE AS CHECKBOX.

SELECTION-SCREEN END OF BLOCK RAD2.

SELECTION-SCREEN BEGIN OF BLOCK RAD3 WITH FRAME TITLE TEXT-003.

PARAMETERS: P_KENAR1 TYPE INT4 MODIF ID G1.

SELECTION-SCREEN END OF BLOCK RAD3 .

SELECTION-SCREEN BEGIN OF BLOCK RAD4 WITH FRAME TITLE TEXT-004.

PARAMETERS: P_KENAR2 TYPE INT4 MODIF ID G2.

SELECTION-SCREEN END OF BLOCK RAD4.

INITIALIZATION.

  P_KARE = ABAP_TRUE.

AT SELECTION-SCREEN OUTPUT.

  IF P_KARE EQ ABAP_TRUE.
    CLEAR: P_KENAR2.
  ENDIF.

  LOOP AT SCREEN.

    IF P_KARE EQ ABAP_TRUE OR P_UCGEN EQ ABAP_TRUE.
      IF SCREEN-GROUP1 EQ 'G2'.
        SCREEN-ACTIVE = 0.
        MODIFY SCREEN.
      ELSEIF SCREEN-GROUP1 EQ 'G1'.
        SCREEN-ACTIVE = 1.
        MODIFY SCREEN.
      ENDIF.
    ELSEIF P_KARE EQ ABAP_FALSE OR P_UCGEN EQ ABAP_FALSE.
      IF SCREEN-GROUP1 EQ 'G2'.
        SCREEN-ACTIVE = 1.
        MODIFY SCREEN.
      ELSEIF SCREEN-GROUP1 EQ 'G1'.
        SCREEN-ACTIVE = 1.
        MODIFY SCREEN.
      ENDIF.
    ENDIF.
  ENDLOOP.


START-OF-SELECTION.


  IF P_KARE EQ ABAP_TRUE.
    GV_ALAN = P_KENAR1 * P_KENAR1.
    GV_CEVRE = P_KENAR1 * 4.
  ELSEIF P_DIKDOR EQ ABAP_TRUE .
    GV_ALAN = P_KENAR1 * P_KENAR2 .
    GV_CEVRE = ( P_KENAR1 * 2 ) + ( P_KENAR2 * 2 ) .
  ELSEIF P_UCGEN EQ ABAP_TRUE.
    GV_ALAN = P_KENAR1 * P_KENAR1 * GV_CONSTANT .
    GV_CEVRE = P_KENAR1 * 3.
  ENDIF.


  IF P_KARE EQ ABAP_TRUE.
    IF P_CEVRE EQ ABAP_TRUE AND P_ALAN EQ ABAP_TRUE.
      WRITE: 'Karenin Alanı:', GV_ALAN, / 'Karenin Çevresi:', GV_CEVRE.
    ELSEIF P_ALAN EQ ABAP_TRUE.
      WRITE: 'Karenin Alanı:', GV_ALAN.
    ELSEIF P_CEVRE EQ ABAP_TRUE.
      WRITE: 'Karenin Çevresi:', GV_CEVRE.

    ENDIF .
  ELSEIF P_DIKDOR EQ ABAP_TRUE.
    IF P_CEVRE EQ ABAP_TRUE AND P_ALAN EQ ABAP_TRUE.
      WRITE: 'Dikdörtgenin Alanı:', GV_ALAN, / 'Dikdörtgenin Çevresi:', GV_CEVRE.
    ELSEIF P_ALAN EQ ABAP_TRUE.
      WRITE: 'Dikdörtgenin Alanı:' , GV_ALAN.
    ELSEIF P_CEVRE EQ ABAP_TRUE.
      WRITE: 'Dikdörtgenin Çevresi:', GV_CEVRE.

    ENDIF.
  ELSEIF P_UCGEN EQ ABAP_TRUE.
    IF P_CEVRE EQ ABAP_TRUE AND P_ALAN EQ ABAP_TRUE.
      WRITE: 'Üçgenin Alanı:', GV_ALAN, / 'Üçgenin Çevresi:', GV_CEVRE.
    ELSEIF P_ALAN EQ ABAP_TRUE.
      WRITE: 'Üçgenin Alanı:', GV_ALAN.
    ELSEIF P_CEVRE EQ ABAP_TRUE.
      WRITE:  'Üçgenin Çevresi:', GV_CEVRE.

    ENDIF.
  ENDIF.
